Custom CSS styles

Test link

  • Passed by "css" option

    <a href="#" id="link">Test link</a>

<script type="text/javascript">
    $('#link').shTooltip({
        css: {
            boxShadow: "0 0 10px #000, inset 0 0 10px #08f",
            background: "#fff",
            padding: "10px",
            borderColor: "#08f",
            'borderRadius': "17px"
        }
    });
</script>

  • Using CSS namespace

    <style type="text/css">
.myNS {
    box-shadow: 0 0 10px #000, inset 0 0 10px #08f;
    background: #fff;
    padding: 10px;
    border-radius: 17px;
    border-color: #08f;
}
</style>

<a href="#" id="link">Test link</a>

<script type="text/javascript">
    $('#link').shTooltip({
        namespace: "myNS"
    });
</script>
